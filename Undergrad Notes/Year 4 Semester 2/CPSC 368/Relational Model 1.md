**Relational Model 1:**
 
CREATE TABLE F  
(f1 INTEGER PRIMARY KEY,  
f2 INTEGER)
 
CREATE TABLE D  
(d1 INTEGER PRIMARY KEY,  
d2 INTEGER)
 
CREATE TABLE ET  
(e1 INTEGER PRIMARY KEY,  
e2 INTEGER,  
f1 INTEGER NOT NULL,  
d1 INTEGER NOT NULL UNIQUE,  
FOREIGN KEY (f1)  
REFERENCES F(f1)  
FOREIGN KEY (d1)  
REFERENCES D(d1))
